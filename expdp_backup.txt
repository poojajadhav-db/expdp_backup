Expdp

1) Table-level_expdp

> Create physical directory 

> create logical directory

create of replace directory exp-table as  '/backup/table_exp';

expdp directory = exp_table dumpfile=table_exp.%u.dmp 
logfile=table_exp.log
tables=u1.testable parallel=2
